    # Actual
    Truck_Capacity=1000
    counts = Data['Hour'].value_counts()/4
    counts = counts.sort_index(axis = 0)
    t=counts.to_numpy()
    Mean1=counts.mean()
    ypoints = t
    TruckNum=round(Mean1/Truck_Capacity)
    #TruckNum=12
    d=pd.DataFrame
    cdf=t-(Truck_Capacity*TruckNum)
    x = np.arange(start=0, stop=20, step=1)
    x[0]=cdf[0]
    U= np.zeros(20)
    CY = np.cumsum(cdf)
    for i in range(0,19):
        x[i+1]=x[i]+cdf[i+1]
        if x[i+1]<0:
            U[i+1]=x[i+1]
            x[i+1]=0
    # Utilization Rate
    d=pd.DataFrame
    d=0
    for i in range(0,19):
        if U[i]<0:
            d=d+U[i]
    Utilization=((Truck_Capacity*TruckNum*20)+d)/(Truck_Capacity*TruckNum*20)*100
    plt.figure(figsize=(30, 10))
    #ax = sns.countplot(x="Hour", hue="Date", data=Data,palette='pastel')
    plt.axhline(y=Mean1, color='r', linestyle='-',linewidth = '7')
    plt.text(0.1, Mean1, "Mean = %s" % round(Mean1), horizontalalignment='left', fontsize=30)
    plt.axhline(y=x.max(), color='r', linestyle=':',linewidth = '5')
    plt.text(0.1, x.max(),  "MAX = %s" % round(x.max()), horizontalalignment='left', fontsize=30)
    plt.text(17, x.max(),  "Trucks = %s" % TruckNum, horizontalalignment='left', fontsize=30)
    plt.text(16, Mean1/2,  "Utilization = %s " % round(Utilization,2), horizontalalignment='left', fontsize=30)
    plt.plot(ypoints, color = 'g', linewidth = '7')
    plt.plot(x, color = 'b', linewidth = '7')
    plt.show()
